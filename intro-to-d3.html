In computer science, declarative programming is a programming paradigm, a style of building the structure and elements of computer programs, that expresses the logic of a computation without describing its control flow.

- Wikipedia article on Declarative Programming

Data-Driven Documents

D3 supports static as well as dynamic properties on DOM elements

body.selectAll("p")
    .style("color", "#AAAAAA");

body.selectAll("p")
    .style("color", function(d, i) {
        return "hsl(" + Math.random() * 360 + ",100%,50%)";
    });

Both Valid

Data

Data-Driven Documents


D3 joins data elements to dom elements with the data function. This is easiest to understand with an example

var home = 
    ["Walter","Skyler", 
     "Hank", "Marie"];
var para =
    body.selectAll("p").data(names);

This has no direct effect. This call to data tells D3 that you want paragraph elements in the document body to be associated with the names in the array. This creates 3 selections that can be manipulated.

Data

A D3 data join creates three selections

  • Enter - representing arriving data elements
  • Exit - representing departing data elements
  • Update - representing existing data elements

Data - Enter

Add a paragraph element for each new data item

   var home = ["Walter","Skyler", "Hank", "Marie"];
     
     var para = body.selectAll("p")
                    .data(names);
     
     para.enter().append("p"); // enter 
   };

Data - Exit

Remove the paragraph item for removed data items

   var home = ["Walter","Skyler", "Hank", "Marie"];
     
     var para = body.selectAll("p")
                    .data(names);
     
     para.exit().remove(); // exit
   };

Data - Update

Set the paragraph text to the data item value
(note the use of a function of the data item as the value)

   var home = ["Walter","Skyler", "Hank", "Marie"];
     
     var para = body.selectAll("p")
                    .data(names);
     
     para.text(function(d) { return d; }); // update
   };

Data - Putting it together

example 3a

<script type="text/javascript">
   var home = ["Walter","Skyler", "Hank", "Marie"];

   function update(names){
     var body=d3.select("body");
     
     var para = body.selectAll("p")
                    .data(names);
     
     para.enter().append("p"); // enter 
     para.exit().remove(); // exit
     para.text(function(d) { return d; }); // update

   };

   function button1_click() {
     update(home);
   }

     body.selectAll("p").data([]).exit().remove();
   };
  </script>

Scalable Vector Graphics

Scalable Vector Graphics (SVG) is an XML-based vector image format for two-dimensional graphics with support for interactivity and animation

- Wikipedia article on SVG

SVGs are hierarchical. Individual elements can have their own style annotations and even motion transformation annotations.

Dynamic SVG with text

function update(names){
     var body=d3.select("body");
     
     // bind our data to SVG Text elements
     var text = svg.selectAll("text").data(names);

     text.enter()
         .append("text") // create the text element
         .attr("x", 10) // set the x as constant
         .attr("y", function(d, i) { return (i + 1) *25;}) // based on index
         .style("font-size", "18") 
         .style("fill", "blue");

     text.text(function(d){return d;});
     text.exit().remove(); // exit
   };

   function button1_click() {
     update(home);
   }
   
   function reset_click() {
   update([]);
   };

Bar Chart Example 1

     var width = 300;
     var height = 350;
     
     var chart = d3.select(".chart")
                   .attr("width", width)
                   .attr("height", height);
     
     var data = [1, 2, 4, 8, 16];
     
     var barWidth = width / data.length;
     
     var bar = chart.selectAll("g").data(data);

Bar Chart Example 2

     bar.enter().append("g")
        .attr("transform", 
              function(d, i) 
              { return "translate(" + i * barWidth + ",0)"; });
     
     // add the rectangle
     bar.append("rect")
        .attr("y", function(d, i) { return height - (d * 20); })
        .attr("height", function(d, i) { return d * 20; })
        .attr("width", barWidth - 1);
     
     // add the label
     bar.append("text")
        .attr("x", barWidth / 2)
        .attr("y", function(d, i) { return height - (d * 20) - 20; })
        .attr("dy", ".75em")
        .style("font-size", "40px")
        .style("fill", "#111111")
        .text(function(d, i) { return d; });
