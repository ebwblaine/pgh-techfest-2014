  <head>
    <meta content="text/html;charset=utf-8" http-equiv="Content-Type">
    <meta content="utf-8" http-equiv="encoding">
    <script type="text/javascript" src="js/d3.v3.min.js"></script>
    <style type="text/css">

     .chart rect {
       fill: steelblue;
     }

     .chart text {
       fill: white;
       font: 10px sans-serif;
       text-anchor: middle;
     }

    </style>
  </head>

  <body>
    <h2>Example 5a</h2>
    <script>
     
     var width = 300;
     var height = 350;
     
     var chart = d3.select("body")
                   .append("svg")
                   .attr("class", "chart")
                   .attr("width", width)
                   .attr("height", height);
     
     var data = [1, 2, 4, 8, 16];
     
     var barWidth = width / data.length;
     
     var bar = chart.selectAll("g")
                    .data(data);

     bar.enter().append("g")
        .attr("transform", 
              function(d, i) 
              { return "translate(" + i * barWidth + ",0)"; });
     
     // add the rectangle
     bar.append("rect")
        .attr("y", function(d, i) { return height - (d * 20); })
        .attr("height", function(d, i) { return d * 20; })
        .attr("width", barWidth - 1);
     
     // add the label
     bar.append("text")
        .attr("x", barWidth / 2)
        .attr("y", function(d, i) { return height - (d * 20) + 15; })
        .style("font-size", "40px")
        .style("fill", "#111111")
        .text(function(d, i) { return d; });

    </script>
  </body>
