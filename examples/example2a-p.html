   function button1_click() {
     var body =d3.select("body"); 

     body.selectAll("p")

       return "hsl(" + Math.random() * 360 + ",100%,50%)";
     });
   };

   function reset_click() {
     var body = d3.select("body"); 
     body.selectAll("p")
         .style("color", null);
   };
